6.3 建堆

# 说明

创建最大堆

# 代码

def build_max_heap(A:List[int]):
    n = len(A)//2
    for i in range(n,-1,-1):
        max_heapify(A,i)

param = [4,1,3,2,16,9,10,14,8,7]
build_max_heap(param)
print(param)
#[16, 14, 10, 8, 7, 9, 3, 2, 4, 1]
