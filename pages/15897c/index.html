    常规配置

    # 配置本地虚拟网卡和ip

    编辑/etc/network/interfaces,添加如下内容

    allow-hotplug tun0
auto tun0
iface tun0 inet manual
    pre-up ip tuntap add tun0 mode tun user root
    pre-up ip addr add 1.1.1.2/32 dev tun0
    up ip link set dev tun0 up
    post-down ip link del dev tun0

    # 创建虚拟2G内存

    fallocate -l 2048M /swapfile
dd if=/dev/zero of=/swapfile bs=1M count=2048
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile
echo "/swapfile swap swap defaults 0 0" | tee -a /etc/fstab
