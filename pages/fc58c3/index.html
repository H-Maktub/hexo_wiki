31.6 元素的幂

# 说明

快速幂取余

# 代码

def power_mod(base, exponent, modulus):
    result = 1
    base = base % modulus
    while exponent > 0:
        if (exponent % 2) == 1:
            result = (result * base) % modulus
        exponent = exponent >> 1
        base = (base * base) % modulus
    return result

# 示例用法
base = 5
exponent = 3
modulus = 7
result = power_mod(base, exponent, modulus)
print(f"{base} 的 {exponent} 次方对 {modulus} 取余数是 {result}")
