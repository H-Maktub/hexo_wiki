    Plex

    Plex (opens new window)

    # 说明

    家庭影音服务

    # ARM需要自己手动生成镜像

    下载Github仓库代码

    #下载代码
git clone https://github.com/plexinc/pms-docker.git
cd pms-docker
# 生成
docker build -t plexinc/pms-docker:latest -f Dockerfile.arm64 .

    # 搭建

    version: "3"
services:
  plex:
    image: plexinc/pms-docker
    restart: always
    network_mode: host
    container_name: plex   
    volumes:   
      - ./config:/config
      - ./transcode:/transcode
      - /mnt/down/download/Plex:/data
    environment: 
      - UMASK=000
      - PUID=0
      - GUID=0
