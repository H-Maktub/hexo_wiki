    迅雷离线下载，邀请码我不是矿神IMNKS

    # 搭建

    version: "3"
services:
    xunlei:
        image: cnk3x/xunlei:latest
        container_name: xunlei
        restart: unless-stopped
        privileged: true
        hostname: mynas
        network_mode: bridge
        ports:
            - 2345:2345
        volumes:
            - /mnt/disk/docker/xunlei:/xunlei/data
            - /mnt/down/xunleidownloads:/xunlei/downloads
        environment:
            - PUID=0
            - PGID=0
            - UMASK=022
            - XL_BA_USER=<用户名>
            - XL_BA_PASSWORD=<密码>
