腾讯云手动挂载对象存储

# 腾讯云相关设置

# 创建子帐号

  1. 创建子帐号，链接https://console.cloud.tencent.com/cam (opens new window)
  2. 相关流程和配置项
    • 选择自定义创建
    • 选择可访问资源并接收消息
    • 选择编程访问
    • 权限选择QcloudCOSDataFullControl(对象存储（COS）数据读、写、删除、列出的访问权限)

# 创建存储桶

  1. 创建存储桶，链接https://console.cloud.tencent.com/lighthouse/cos/index?rid=1 (opens new window)
  2. 添加用户权限
    • 子帐号
    • 完全控制

# 配置系统

  1. 安装相关软件
apt install rclone fuse3 -y
  1. 配置文件/root/.config/rclone/rclone.conf
[cos]
type = s3
provider = TencentCOS
env_auth = false
access_key_id = <帐号key>
secret_access_key = <帐号secret>
endpoint = cos.ap-guangzhou.myqcloud.com
acl = default
storage_class = default
  1. 创建挂载点
mkdir /mnt/cosfs
mkdir /mnt/cosfs-cache
# 测试
rclone mount cos:/<存储桶的名称> /mnt/cosfs --cache-dir /mnt/cosfs-cache --vfs-cache-mode writes
  1. 配置系统服务/etc/systemd/system/cosfs.service
[Unit]
Description=Rclone Mount CosFS Service
After=network.target

[Service]
Type=simple
User=root
Restart=on-failure
RestartSec=5s
ExecStart=/usr/bin/rclone mount cos:/<存储桶的名称> /mnt/cosfs --cache-dir /mnt/cosfs-cache --vfs-cache-mode writes

[Install]
WantedBy=multi-user.target
  1. 启动服务
systemctl enable cosfs
systemctl start cosfs
systemctl status cosfs
