    # 说明

    电子书管理

    # 搭建

    version: '3.3'
services:
    calibre:
        image: linuxserver/calibre-web:latest
        container_name: calibre-web
        volumes:
            - /mnt/disk/docker/calibre/config:/config
            - /mnt/disk/docker/calibre/books:/books
        ports:
            - 8083:8083
        environment:
            - PUID=0
            - PGID=0
            - UMASK=022
            - TZ=Asia/Chongqing
        restart: always
